MongoDB Connection mongoose.connect(process.env.MONGODB_URI, { useNewUrlParser: true, useUnifiedTopology: true }) .then(() => { console.log("âœ… MongoDB Connected"); }) .catch(err => { console.error("âŒ MongoDB Connection Error:", err.message); }); // âœ… 2.
